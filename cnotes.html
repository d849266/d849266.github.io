01. Declare function returning pointer to array size N of type T 

    T (* f(void))[N]
    {
      static T a[N];
      return &a;
    }

Alternative

    typedef T array[N];

    array* f(void)
    {
      static array a;
      return &a;
    }

Usage

    (*f())[i] = 42;
